const App = () => {
  const [task, setTask] = useState("");
  const [detail, setDetail] = useState("");
  const [tasks, setTasks] = useState([]);

  const handleSubmit = (e) => {
    e.preventDefault();
    const newTask = { task, detail };
    setTasks([...tasks, newTask]);
    setTask("");
    setDetail("");
  };

  return (
    <div className="w-screen h-screen flex flex-col md:flex-row overflow-hidden">
      
      {/* Left Section (Form) */}
      <div className="w-full md:w-1/2 h-1/2 md:h-full bg-gradient-to-br from-indigo-600 via-purple-600 to-pink-500 flex justify-center items-center p-6">
        <div className="w-full max-w-md bg-white/10 backdrop-blur-md rounded-2xl p-8 text-white shadow-2xl">
          <h2 className="text-3xl font-bold mb-6 text-center">üìù Add Your Task</h2>
          <form onSubmit={handleSubmit} className="flex flex-col gap-4">
            <input
              type="text"
              placeholder="Enter task title"
              value={task}
              onChange={(e) => setTask(e.target.value)}
              className="p-3 rounded-lg border-none focus:outline-none focus:ring-2 focus:ring-pink-300 text-gray-800"
              required
            />
            <textarea
              placeholder="Enter task details"
              value={detail}
              onChange={(e) => setDetail(e.target.value)}
              rows="3"
              className="p-3 rounded-lg border-none focus:outline-none focus:ring-2 focus:ring-pink-300 text-gray-800"
              required
            ></textarea>
            <button
              type="submit"
              className="bg-white text-purple-700 font-semibold py-2 rounded-lg hover:bg-purple-100 transition-all"
            >
              Add Task
            </button>
          </form>
        </div>
      </div>

      {/* Right Section (Tasks in flexible cards) */}
      <div className="w-full md:w-1/2 h-1/2 md:h-full bg-gray-50 flex flex-col p-8 overflow-y-auto">
        <h2 className="text-3xl font-bold text-gray-800 mb-6 text-center">üìã Task List</h2>

        {tasks.length === 0 ? (
          <div className="flex justify-center items-center h-full">
            <p className="text-gray-500 text-lg">No tasks added yet.</p>
          </div>
        ) : (
          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-6 auto-rows-auto">
            {tasks.map((t, idx) => (
              <div
                key={idx}
                className="bg-white shadow-md border border-gray-200 rounded-xl p-5 hover:shadow-xl transition-all duration-300 flex flex-col"
              >
                <h3 className="text-lg font-semibold text-purple-700 mb-2 break-words">
                  {t.task}
                </h3>
                <p className="text-gray-600 break-words">{t.detail}</p>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
};
